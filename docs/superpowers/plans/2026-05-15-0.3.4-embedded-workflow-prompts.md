# 0.3.4 Embedded Workflow Prompts Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Detect agent instructions embedded in GitHub Actions workflows, starting with Claude Code Action prompt inputs, and surface them in manifests and conservative findings.

**Architecture:** Add a focused workflow prompt extractor that parses already-read workflow YAML files into source-backed facts. Wire those facts into `RepoFacts`, manifests, and pure checks. Keep CLI behavior unchanged except for additional manifest facts and findings.

**Tech Stack:** TypeScript, Vitest, existing `yaml` dependency, existing scanner pipeline, existing text/JSON/SARIF reporters.

---

## Scope

### In scope

- Extract workflow-embedded Claude Code Action prompt inputs from `.github/workflows/*.yml` and `.yaml`.
- Support these prompt surfaces:
  - `custom_instructions`
  - `claude_args` with `--system-prompt <value>` or `--system-prompt=<value>`
  - `prompt`
  - `direct_prompt`
  - `claude_args` with `--append-system-prompt <value>` or `--append-system-prompt=<value>`
- Add `workflowPrompts` to `RepoFacts` and manifest output.
- Add conservative findings:
  - `unsafe-workflow-prompt`
  - `hidden-workflow-prompt`
- Preserve source evidence for every prompt and finding.
- Update README, CHANGELOG, roadmap, architecture docs, and release smoke notes.

### Out of scope

- Non-Claude workflow actions.
- Full shell parsing for arbitrary `claude_args`.
- Semantic conflict detection between workflow prompts and repo docs.
- Runtime and tool-version drift checks.
- Broader verification command conflict checks across instruction surfaces.
- Rule-quality, safety, and workflow hygiene checks.
- Network/API lookups.
- MCP context gate.
- AI remediation.

Create `src/extractors/workflowPrompts.ts`:

```ts
import { parse } from 'yaml';
import type { RawFile, WorkflowPrompt, WorkflowPromptKind } from '../core/types.js';

const promptInputKinds: Record<string, WorkflowPromptKind> = {
  custom_instructions: 'custom-instructions',
  prompt: 'prompt',
  direct_prompt: 'direct-prompt'
};

const claudeArgPromptKinds: Record<string, WorkflowPromptKind> = {
  '--system-prompt': 'system-prompt',
  '--append-system-prompt': 'append-system-prompt'
};

const claudeActions = new Set(['anthropics/claude-code-action', 'anthropics/claude-code-base-action']);

export function extractWorkflowPrompts(files: RawFile[]): WorkflowPrompt[] {
  return files.filter((file) => isGitHubWorkflow(file.path)).flatMap((file) => extractFromWorkflow(file));
}

function extractFromWorkflow(file: RawFile): WorkflowPrompt[] {
  const document = parseWorkflow(file.content);
  if (!isRecord(document) || !isRecord(document.jobs)) {
    return [];
  }

  const prompts: WorkflowPrompt[] = [];
  for (const job of Object.values(document.jobs)) {
    if (!isRecord(job) || !Array.isArray(job.steps)) {
      continue;
    }

    for (const step of job.steps) {
      if (!isRecord(step) || typeof step.uses !== 'string' || !isClaudeAction(step.uses) || !isRecord(step.with)) {
        continue;
      }

      for (const [key, kind] of Object.entries(promptInputKinds)) {
        const value = step.with[key];
        if (typeof value === 'string' && value.trim() !== '') {
          prompts.push({ kind, action: step.uses, value: value.trim(), source: sourceForKey(file, key) });
        }
      }

      if (typeof step.with.claude_args === 'string') {
        for (const parsedPrompt of parseClaudeArgPrompts(step.with.claude_args)) {
          prompts.push({ kind: parsedPrompt.kind, action: step.uses, value: parsedPrompt.value, source: sourceForClaudeArgPrompt(file, parsedPrompt.flag) });
        }
      }
    }
  }

  return prompts;
}

function parseWorkflow(content: string): unknown {
  try {
    return parse(content);
  } catch {
    return undefined;
  }
}

function parseClaudeArgPrompts(value: string): Array<{ kind: WorkflowPromptKind; flag: string; value: string }> {
  const prompts: Array<{ kind: WorkflowPromptKind; flag: string; value: string }> = [];
  for (const [flag, kind] of Object.entries(claudeArgPromptKinds)) {
    const escapedFlag = flag.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const inline = value.match(new RegExp(`${escapedFlag}=("[^"]+"|'[^']+'|\\S+)`));
    const spaced = value.match(new RegExp(`${escapedFlag}\\s+("[^"]+"|'[^']+'|[^\\s][^\\n]*)`));
    const raw = inline?.[1] ?? spaced?.[1];
    if (raw) {
      prompts.push({ kind, flag, value: stripQuotes(raw.trim()) });
    }
  }
  return prompts;
}

function sourceForKey(file: RawFile, key: string) {
  return sourceForLine(file, (line) => line.trim().startsWith(`${key}:`));
}

function sourceForClaudeArgPrompt(file: RawFile, flag: string) {
  return sourceForLine(file, (line) => line.includes(flag));
}

function sourceForLine(file: RawFile, predicate: (line: string) => boolean) {
  const lines = file.content.split('\n');
  const index = lines.findIndex((line) => predicate(line.replace(/\r$/, '')));
  const line = index >= 0 ? lines[index].replace(/\r$/, '') : '';
  return { file: file.path, line: index >= 0 ? index + 1 : undefined, text: line.trim() };
}

function stripQuotes(value: string): string {
  if ((value.startsWith('"') && value.endsWith('"')) || (value.startsWith("'") && value.endsWith("'"))) {
    return value.slice(1, -1);
  }
  return value;
}

function isClaudeAction(action: string): boolean {
  const name = action.split('@')[0].toLowerCase();
  return claudeActions.has(name);
}

function isGitHubWorkflow(path: string): boolean {
  return /^\.github\/workflows\/[^/]+\.ya?ml$/i.test(path);
}

function isRecord(value: unknown): value is Record<string, unknown> {
  return typeof value === 'object' && value !== null && !Array.isArray(value);
}
```

The implementation sketch above is intentionally minimal, but the final implementation must scope source lookup to the matched workflow step. Do not use an unbounded whole-file `findIndex()` when a workflow contains multiple Claude steps with the same prompt key or flag. If lint or tests reveal that the simple parser captures too much for unquoted `--system-prompt` or `--append-system-prompt`, narrow it. Prefer skipping ambiguous values over guessing.

Legacy note: `custom_instructions` and `direct_prompt` are deprecated Claude Code Action v0 inputs, but 0.3.4 should still extract them because real repositories may not have migrated. Current v1 workflows should primarily be covered by `prompt` and `claude_args --system-prompt`.

```ts
import type { Check, Finding } from '../core/types.js';

const unsafePatterns = [
  /\bskip tests\b/i,
  /\bignore lint\b/i,
  /--no-verify/i,
  /\bforce push\b/i,
  /git push --force/i
];

const negatedPatterns = [
  /\bdo not skip tests\b/i,
  /\bdon't skip tests\b/i,
  /\bnever skip tests\b/i,
  /\bnever (?:use )?.*--no-verify\b/i,
  /\bdo not .*--no-verify\b/i,
  /\bdon't .*--no-verify\b/i,
  /\bavoid .*--no-verify\b/i,
  /\bdo not .*force push\b/i,
  /\bdon't .*force push\b/i,
  /\bnever .*force push\b/i,
  /\bavoid .*force push\b/i
];

export const unsafeWorkflowPromptCheck: Check = {
  id: 'unsafe-workflow-prompt',
  run({ facts }) {
    return facts.workflowPrompts.flatMap((prompt): Finding[] => {
      if (!unsafePatterns.some((pattern) => pattern.test(prompt.value)) || negatedPatterns.some((pattern) => pattern.test(prompt.value))) {
        return [];
      }
      return [{
        id: 'unsafe-workflow-prompt',
        title: 'Workflow-embedded agent prompt includes unsafe guidance',
        category: 'agent-instructions',
        severity: 'warning',
        confidence: 'medium',
        primarySource: prompt.source,
        evidence: [{
          kind: 'workflow-prompt',
          message: `${prompt.source.file}:${prompt.source.line ?? 1} embeds ${prompt.kind} for ${prompt.action}.`,
          source: prompt.source
        }],
        suggestion: 'Move safe agent guidance into repo-visible instructions and replace bypass guidance with explicit verification expectations.'
      }];
    });
  }
};

export const hiddenWorkflowPromptCheck: Check = {
  id: 'hidden-workflow-prompt',
  run({ facts }) {
    if (facts.workflowPrompts.length === 0 || facts.agentInstructionDocs.length > 0) {
      return [];
    }
    const prompt = facts.workflowPrompts[0];
    return [{
      id: 'hidden-workflow-prompt',
      title: 'Agent prompt exists only inside a workflow',
      category: 'agent-instructions',
      severity: 'info',
      confidence: 'high',
      primarySource: prompt.source,
      evidence: [{
        kind: 'workflow-prompt',
        message: `${prompt.source.file}:${prompt.source.line ?? 1} embeds ${prompt.kind}, but no repo-visible agent instruction file was found.`,
        source: prompt.source
      }],
      suggestion: 'Add or reference canonical agent instructions in AGENTS.md or CLAUDE.md so local agents and humans see the same guidance.'
    }];
  }
};
```
