# 0.3.2 Scoped Effective Context Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add path-scoped effective instruction context, opt-in workspace parent inheritance, Cursor scoped rule support, and conservative scoped-rule findings.

**Architecture:** Keep `readWorkspace -> extractors -> RepoFacts -> checks/reporters`. Add a pure `effectiveContext` resolver consumed by manifest and workspace scan adapters. Add scoped-rule extraction metadata and checks without changing default scan behavior.

**Tech Stack:** TypeScript, Commander, Vitest, fast-glob-compatible path/glob utilities, existing YAML/frontmatter parsing patterns.

- [ ] **Step 4: Implement resolver**

Create `src/core/effectiveContext.ts`:

```ts
import { minimatch } from 'minimatch';
import type { AgentInstructionDocFact, EffectiveContext, EffectiveInstructionFile, RepoFacts } from './types.js';

export type EffectiveContextOptions = {
  targetPath?: string;
  includeInherited?: boolean;
  parentInstructionFiles?: AgentInstructionDocFact[];
};

export function resolveEffectiveContext(facts: RepoFacts, options: EffectiveContextOptions = {}): EffectiveContext {
  const targetPath = normalizePath(options.targetPath ?? '');
  const inheritedDocs = options.includeInherited ? options.parentInstructionFiles ?? [] : [];
  const localEntries = facts.agentInstructionDocs
    .filter((doc) => appliesToTarget(doc, targetPath))
    .map((doc) => toEffectiveInstructionFile(doc, false, appliesBecause(doc, targetPath)));
  const inheritedEntries = inheritedDocs.map((doc) => toEffectiveInstructionFile(doc, true, 'inherited from workspace parent because --inherit-parent-instructions is enabled'));

  return {
    targetPath: targetPath || undefined,
    instructionFiles: dedupeEffectiveInstructions([...inheritedEntries, ...localEntries])
  };
}

function appliesToTarget(doc: AgentInstructionDocFact, targetPath: string): boolean {
  if (!targetPath) {
    return doc.scope === 'repo';
  }

  if (doc.scope === 'repo') {
    return true;
  }

  if (doc.appliesTo?.some((pattern) => safeMinimatch(targetPath, pattern))) {
    return true;
  }

  if (doc.tool === 'cursor' && doc.metadata?.alwaysApply === true) {
    return true;
  }

  if (doc.tool === 'cursor') {
    return false;
  }

  if (doc.path.endsWith('/AGENTS.md')) {
    const directory = doc.path.slice(0, -'AGENTS.md'.length);
    return targetPath.startsWith(directory);
  }

  return false;
}

function appliesBecause(doc: AgentInstructionDocFact, targetPath: string): string {
  if (doc.scope === 'repo') {
    return 'repo-level instruction applies to all paths under the requested root';
  }

  const matchingGlob = doc.appliesTo?.find((pattern) => safeMinimatch(targetPath, pattern));
  if (matchingGlob) {
    return `target path matches scoped pattern ${matchingGlob}`;
  }

  if (doc.tool === 'cursor' && doc.metadata?.alwaysApply === true) {
    return 'Cursor rule has alwaysApply=true';
  }

  if (doc.path.endsWith('/AGENTS.md')) {
    const directory = doc.path.slice(0, -'AGENTS.md'.length);
    return `target path is under ${directory}`;
  }

  return 'instruction surface applies to the requested target path';
}

function toEffectiveInstructionFile(doc: AgentInstructionDocFact, inherited: boolean, appliesBecause: string): EffectiveInstructionFile {
  return {
    path: doc.path,
    type: doc.tool,
    scope: doc.scope,
    appliesTo: doc.appliesTo,
    metadata: doc.metadata,
    source: doc.source,
    inherited,
    inheritedFrom: doc.inheritedFrom,
    displayPath: doc.displayPath,
    appliesBecause
  };
}

function dedupeEffectiveInstructions(entries: EffectiveInstructionFile[]): EffectiveInstructionFile[] {
  const seen = new Set<string>();
  return entries.filter((entry) => {
    const key = `${entry.inherited ? 'inherited' : 'local'}:${entry.path}`;
    if (seen.has(key)) {
      return false;
    }

    seen.add(key);
    return true;
  });
}

function safeMinimatch(path: string, pattern: string): boolean {
  try {
    return minimatch(path, normalizePath(pattern), { dot: true });
  } catch {
    return false;
  }
}

function normalizePath(path: string): string {
  return path.replace(/\\/g, '/').replace(/^\.\//, '');
}
```

```ts
import { parse } from 'yaml';

function cursorMetadata(content: string): { appliesTo?: string[]; metadata?: Record<string, string | string[] | boolean> } {
  const match = /^---\n([\s\S]*?)\n---/.exec(content);
  if (!match) {
    return {};
  }

  const parsed = safeParseFrontmatter(match[1]);
  if (!parsed) {
    return {};
  }

  const globs = stringList(parsed.globs);
  const paths = stringList(parsed.paths);
  const appliesTo = [...globs, ...paths];
  const metadata: Record<string, string | string[] | boolean> = { scopedRule: true };

  if (globs.length > 0) {
    metadata.globs = globs;
  }

  if (paths.length > 0) {
    metadata.paths = paths;
  }

  if (typeof parsed.alwaysApply === 'boolean') {
    metadata.alwaysApply = parsed.alwaysApply;
  }

  return {
    appliesTo: appliesTo.length > 0 ? appliesTo : undefined,
    metadata: Object.keys(metadata).length > 0 ? metadata : undefined
  };
}

function safeParseFrontmatter(source: string): Record<string, unknown> | undefined {
  try {
    const parsed = parse(source);
    return typeof parsed === 'object' && parsed !== null && !Array.isArray(parsed) ? parsed as Record<string, unknown> : undefined;
  } catch {
    return undefined;
  }
}

function stringList(value: unknown): string[] {
  if (typeof value === 'string') {
    return [value];
  }

  if (Array.isArray(value)) {
    return value.filter((entry): entry is string => typeof entry === 'string');
  }

  return [];
}
```

```ts
export function normalizeRootContainedPath(root: string, inputPath: string): string {
  const resolvedRoot = resolve(root);
  const resolvedTarget = resolve(resolvedRoot, inputPath);
  const rawRelativePath = relative(resolvedRoot, resolvedTarget);

  if (
    rawRelativePath === '..' ||
    rawRelativePath.startsWith('../') ||
    rawRelativePath.startsWith('..\\') ||
    isAbsolute(rawRelativePath)
  ) {
    throw new Error('--path must stay inside --root');
  }

  return rawRelativePath.replace(/\\/g, '/') || '.';
}
```

```ts
import { minimatch } from 'minimatch';
import type { Check, Finding } from '../core/types.js';

export const scopedRulesCheck: Check = {
  id: 'scoped-rules',
  run(context) {
    const filePaths = context.facts.filePaths;
    const findings: Finding[] = [];

    for (const doc of context.facts.agentInstructionDocs.filter((entry) => entry.tool === 'cursor' && entry.appliesTo && entry.appliesTo.length > 0)) {
      for (const pattern of doc.appliesTo ?? []) {
        if (!isValidGlob(pattern)) {
          findings.push({
            id: 'invalid-scoped-rule-glob',
            title: 'Scoped instruction rule has invalid glob syntax',
            category: 'context-scope',
            severity: 'warning',
            confidence: 'high',
            primarySource: doc.source,
            evidence: [{ kind: 'scoped-rule', message: `Invalid scoped pattern: ${pattern}`, source: doc.source }],
            suggestion: 'Fix the scoped rule glob so Dr. Context and the agent can determine where it applies.'
          });
          continue;
        }

        const matches = filePaths.filter((filePath) => minimatch(filePath, pattern, { dot: true }));
        if (matches.length === 0) {
          findings.push({
            id: 'scoped-rule-matches-no-files',
            title: 'Scoped instruction rule matches no files',
            category: 'context-scope',
            severity: 'info',
            confidence: 'medium',
            primarySource: doc.source,
            evidence: [{ kind: 'scoped-rule', message: `Pattern ${pattern} did not match any scanned files.`, source: doc.source }],
            suggestion: 'Update the scoped pattern or remove the rule if it is stale.'
          });
        }

        if (matches.length > Math.max(20, filePaths.length * 0.8) && filePaths.length > 10) {
          findings.push({
            id: 'scoped-rule-too-broad',
            title: 'Scoped instruction rule matches most scanned files',
            category: 'context-scope',
            severity: 'info',
            confidence: 'low',
            primarySource: doc.source,
            evidence: [{ kind: 'scoped-rule', message: `Pattern ${pattern} matched ${matches.length} of ${filePaths.length} scanned files.`, source: doc.source }],
            suggestion: 'Narrow the scoped pattern if the rule is meant for a specific subtree.'
          });
        }
      }
    }

    return findings;
  }
};

function isValidGlob(pattern: string): boolean {
  try {
    minimatch('', pattern);
    return hasBalancedGlobDelimiters(pattern);
  } catch {
    return false;
  }
}

function hasBalancedGlobDelimiters(pattern: string): boolean {
  const pairs: Array<[string, string]> = [['[', ']'], ['{', '}']];
  return pairs.every(([open, close]) => pattern.split(open).length === pattern.split(close).length);
}
```

- Run a spec compliance review subagent.
- Run a code quality/noise review subagent.
- Fix any required issues in a follow-up commit before starting the next task.

Before release:

- Run the full verification gate from Task 7.
- Confirm `git status --short --branch`.
- Confirm no private dogfood paths or secrets were introduced.
