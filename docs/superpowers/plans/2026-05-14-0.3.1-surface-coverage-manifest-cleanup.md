# 0.3.1 Surface Coverage + Manifest Cleanup Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add official agent instruction surfaces, broader repo facts, and cleaner CI command classification so `drctx manifest` reflects real agent context without shell-plumbing noise.

**Architecture:** Keep the existing `readWorkspace -> extractors -> RepoFacts -> checks/reporters` pipeline. Add a central instruction-surface registry, explicit repo fact types, and a CI command classifier. Checks stay pure; 0.3.1 mostly expands facts and docs without adding noisy findings.

**Tech Stack:** TypeScript, Vitest, fast-glob, YAML parsing already used through `yaml`, Commander CLI.

---

## Scope

This plan implements the `0.3.1` slice from `docs/superpowers/specs/2026-05-14-roadmap-implementation-design.md`.

Included:

- Official instruction surfaces: Copilot, Cursor MDC, Gemini, AGENT_GUIDE-style files, detection entries for Windsurf/Continue/Aider/Cody.
- Repo fact expansion: `Makefile`, `justfile`, `Taskfile.yml`, README command blocks, `.nvmrc`, `.node-version`, package `engines`, GitHub setup action versions.
- Manifest cleanup: classify CI commands and exclude shell plumbing from canonical verification context.
- README support matrix and 0.3.1 changelog entry.

Excluded until later slices:

- `manifest --path`.
- parent/child inheritance.
- workflow embedded Claude prompt scanning.
- runtime/package-manager drift findings.
- baseline/config.
- MCP.

```ts
import type { AgentInstructionScope, AgentInstructionTool } from '../core/types.js';

export type InstructionSurfaceDefinition = {
  id: string;
  tool: AgentInstructionTool;
  scope: AgentInstructionScope;
  patterns: RegExp[];
  globs: string[];
};

export const instructionSurfaces: InstructionSurfaceDefinition[] = [
  { id: 'agents-root', tool: 'agents', scope: 'repo', patterns: [/^AGENTS\.md$/i], globs: ['AGENTS.md'] },
  { id: 'agents-nested', tool: 'agents', scope: 'nested', patterns: [/^.+\/AGENTS\.md$/i], globs: ['**/AGENTS.md'] },
  { id: 'claude', tool: 'claude', scope: 'repo', patterns: [/^CLAUDE\.md$/i], globs: ['CLAUDE.md'] },
  { id: 'cursor-legacy', tool: 'cursor', scope: 'repo', patterns: [/^\.cursorrules$/i], globs: ['.cursorrules'] },
  { id: 'cursor-rules', tool: 'cursor', scope: 'nested', patterns: [/^\.cursor\/rules\/.+\.(?:md|mdc)$/i], globs: ['.cursor/rules/**/*.{md,mdc}'] },
  { id: 'copilot-repo', tool: 'copilot', scope: 'repo', patterns: [/^\.github\/copilot-instructions\.md$/i], globs: ['.github/copilot-instructions.md'] },
  { id: 'copilot-path', tool: 'copilot', scope: 'path', patterns: [/^\.github\/instructions\/.+\.instructions\.md$/i], globs: ['.github/instructions/**/*.instructions.md'] },
  { id: 'gemini', tool: 'gemini', scope: 'repo', patterns: [/^GEMINI\.md$/i], globs: ['GEMINI.md'] },
  { id: 'agent-guide', tool: 'unknown', scope: 'repo', patterns: [/^AGENT_GUIDE\.md$/i], globs: ['AGENT_GUIDE.md'] },
  { id: 'windsurf', tool: 'unknown', scope: 'repo', patterns: [/^\.windsurfrules$/i], globs: ['.windsurfrules'] },
  { id: 'continue', tool: 'unknown', scope: 'nested', patterns: [/^\.continue\/rules\/.+\.(?:md|mdc)$/i], globs: ['.continue/rules/**/*.{md,mdc}'] },
  { id: 'aider', tool: 'unknown', scope: 'repo', patterns: [/^\.aider\.conf\.ya?ml$/i], globs: ['.aider.conf.yml', '.aider.conf.yaml'] },
  { id: 'cody', tool: 'unknown', scope: 'nested', patterns: [/^\.sourcegraph\/cody\/.+\.md$/i], globs: ['.sourcegraph/cody/**/*.md'] }
];

export const instructionSurfaceGlobs = [...new Set(instructionSurfaces.flatMap((surface) => surface.globs))];

export function getInstructionSurfaceForPath(path: string): InstructionSurfaceDefinition | undefined {
  const normalized = path.replace(/\\/g, '/');
  return instructionSurfaces.find((surface) => surface.patterns.some((pattern) => pattern.test(normalized)));
}
```

```ts
import type { BuildTargetFact, RawFile } from '../core/types.js';

export function extractBuildTargets(files: RawFile[]): BuildTargetFact[] {
  return files.flatMap((file) => {
    if (/^(?:Makefile|makefile)$/i.test(file.path)) return extractMakeTargets(file);
    if (/^(?:justfile|Justfile)$/.test(file.path)) return extractJustTargets(file);
    if (/^Taskfile\.ya?ml$/i.test(file.path)) return extractTaskfileTargets(file);
    return [];
  });
}

function extractMakeTargets(file: RawFile): BuildTargetFact[] {
  return file.content.split('\n').flatMap((line, index) => {
    const match = line.match(/^([A-Za-z0-9_.-]+):(?:\s|$)/);
    if (!match || match[1].startsWith('.')) return [];
    return [{ tool: 'make', name: match[1], source: { file: file.path, line: index + 1, text: line } }];
  });
}

function extractJustTargets(file: RawFile): BuildTargetFact[] {
  return file.content.split('\n').flatMap((line, index) => {
    const match = line.match(/^([A-Za-z0-9_.-]+)(?:\s[^:]*)?:\s*$/);
    if (!match) return [];
    return [{ tool: 'just', name: match[1], source: { file: file.path, line: index + 1, text: line } }];
  });
}

function extractTaskfileTargets(file: RawFile): BuildTargetFact[] {
  const lines = file.content.split('\n');
  const tasksLine = lines.findIndex((line) => /^tasks:\s*$/.test(line));
  if (tasksLine === -1) return [];

  return lines.slice(tasksLine + 1).flatMap((line, offset) => {
    const match = line.match(/^\s{2}([A-Za-z0-9_.-]+):\s*$/);
    if (!match) return [];
    return [{ tool: 'taskfile', name: match[1], source: { file: file.path, line: tasksLine + offset + 2, text: line.trim() } }];
  });
}
```

- [ ] **Step 5: Implement runtime version extractor**

Create `src/extractors/runtimeVersions.ts`:

```ts
import type { RawFile, RuntimeVersionFact } from '../core/types.js';

export function extractRuntimeVersions(files: RawFile[]): RuntimeVersionFact[] {
  return files.flatMap((file) => {
    if (file.path === '.nvmrc') return simpleNodeVersion(file, 'nvmrc');
    if (file.path === '.node-version') return simpleNodeVersion(file, 'node-version');
    if (file.path === 'package.json') return packageEngineVersion(file);
    if (/^\.github\/workflows\/[^/]+\.ya?ml$/i.test(file.path)) return workflowNodeVersions(file);
    return [];
  });
}

function simpleNodeVersion(file: RawFile, kind: 'nvmrc' | 'node-version'): RuntimeVersionFact[] {
  const version = file.content.split('\n')[0]?.trim();
  return version ? [{ runtime: 'node', version, kind, source: { file: file.path, line: 1, text: version } }] : [];
}

function packageEngineVersion(file: RawFile): RuntimeVersionFact[] {
  try {
    const parsed = JSON.parse(file.content) as { engines?: { node?: string } };
    const version = parsed.engines?.node;
    return version ? [{ runtime: 'node', version, kind: 'package-engines', source: { file: file.path, line: findLine(file.content, '"node"'), text: version } }] : [];
  } catch {
    return [];
  }
}

function workflowNodeVersions(file: RawFile): RuntimeVersionFact[] {
  const lines = file.content.split('\n');
  return lines.flatMap((line, index) => {
    const match = line.match(/^\s*node-version:\s*['"]?([^'"\s]+)['"]?\s*$/);
    if (!match) return [];
    return [{ runtime: 'node', version: match[1], kind: 'github-actions', source: { file: file.path, line: index + 1, text: line.trim() } }];
  });
}

function findLine(content: string, needle: string): number {
  const index = content.split('\n').findIndex((line) => line.includes(needle));
  return index === -1 ? 1 : index + 1;
}
```

```ts
import type { CiCommandClassification } from '../core/types.js';

export function classifyCiCommand(command: string): CiCommandClassification {
  const trimmed = command.trim();

  if (/^(?:if\b|then\b|else\b|elif\b|fi\b|for\b|while\b|do\b|done\b|case\b|esac\b|exit\b|return\b|set\s+)/.test(trimmed)) {
    return 'shell-control';
  }

  }

  if (/\b(?:npm\s+publish|pnpm\s+publish|yarn\s+npm\s+publish)\b/.test(trimmed)) {
    return 'publish';
  }

  if (/\b(?:setup-node|setup-python|pnpm\/action-setup|actions\/setup-node)@/.test(trimmed)) {
    return 'setup';
  }

  if (/\b(?:npm|pnpm|yarn|bun|corepack\s+pnpm)\s+(?:test|run\s+(?:test|lint|typecheck|build|check|pack:dry-run))\b/.test(trimmed)) {
    return 'verification';

}
```
